# Austro Prussian War Escalation
country_event = {
	id = trigger.1
	title = trigger.1.t
	desc = trigger.1.d
	picture = GFX_report_event_diplomatic_message

	is_triggered_only = no
	
	fire_only_once = yes
	
	hidden = yes
	
	trigger = {
		AND = {
			has_global_flag = austro_prussian_war_escalation_flag

				has_global_flag = fpw_war_escalation
			}
		}
	}

	option = {
		name = trigger.1.a

        set_global_flag = fpw_war_escalation
		news_event = news_fpw.1
	}
}

# austria asks for peace(battle of koniggratz)
country_event = {
	id = trigger.2
	title = trigger.2.t
	desc = trigger.2.d
	picture = GFX_report_event_diplomatic_message

	is_triggered_only = no
	
	fire_only_once = yes
	
	hidden = yes
	
	trigger = {
		AND = {
			PRS = {
				has_war_with = AUS
			}
			AUS = {
				OR = {
					AND = {
						is_ai = yes
						NOT = {
							controls_province = 583
						}
					}
					# Try keeping historical
					AND = {
						surrender_progress > 0.5
						is_ai = no
						NOT = {
							controls_province = 583
						}
					}
				}
			}
			NOT = {
				has_global_flag = fpw_austrian_peace
			}
		}
	}

	option = {
		name = trigger.2.a

        set_global_flag = fpw_austrian_peace
		PRS = {
			country_event = austro_prussian_fpw.2
		}
	}
}
