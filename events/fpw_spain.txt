# Invite to be king
country_event = {
	id = spain_fpw.2
	title = spain_fpw.2.t
	desc = spain_fpw.2.d
	picture = GFX_report_event_generic_handshake
	
	is_triggered_only = yes

	# Accept
	option = {
		name = spain_fpw.2.a
			ai_chance = {
			base = 10 
			modifier = {
				add = 10 
				has_opinion = {
					target = SPR
					value > 10
				}
			}
			modifier = {
				add = 20
				has_opinion = {
					target = SPR
					value > 20
				}
			}
			modifier = {
				add = 30 
				has_opinion = {
					target = SPR
					value > 30
				}
			}
			modifier = {
				factor = 0
				AND = {
					is_historical_focus_on = yes
					original_tag = PRS
				}
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = FRA
				}
			}
			SPR = {
				set_country_flag = SPR_french_king
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = ENG
				}
			}
			SPR = {
				set_country_flag = SPR_british_queen
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = ITA
				}
			}
			SPR = {
				set_country_flag = SPR_italian_king
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = PRS
				}
			}
			PRS = {
				complete_national_focus = PRS_leopold_on_the_spanish_thrown
			}
			SPR = {
				set_country_flag = SPR_prussian_king
			}
		}
		SPR = {
			country_event = spain_fpw.3
		}
	}

	# Decline
	option = {
		name = spain_fpw.2.b
		ai_chance = {
			base = 10 
			modifier = {
				factor = 5
				has_opinion = {
					target = SPR
					value < 1
				}
			}
			modifier = {
				factor = 0
				AND = {
					is_historical_focus_on = yes
					NOT = {
						original_tag = PRS
					}
				}
			}
			modifier = {
				add = 100
				AND = {
					is_historical_focus_on = yes
					original_tag = PRS
				}
			}
		}
		if = {
			limit = {
				original_tag = PRS
			}
			set_global_flag = fpw_declined_leopold
		}
		SPR = {
			country_event = spain_fpw.4
		}
	}
}
