scripted_gui = {

	fuhrer_mana = {
		context_type = player_context
		
		window_name = "infamy_container"
		
		visible = {
            always = yes
		}

		triggers = {
			infamy_icon_regular_visible = {
				check_variable = {
					var = THIS.infamy
					value = 25
					compare = less_than
				}
			}
			
			infamy_icon_infamous_visible = {
				check_variable = {
					var = THIS.infamy
					value = 25
					compare = greater_than_or_equals 
				}
				check_variable = {
					var = THIS.infamy
					value = 50
					compare = less_than
				}
			}

			infamy_icon_notorious_visible = {
				check_variable = {
					var = THIS.infamy
					value = 50
					compare = greater_than_or_equals 
				}
				check_variable = {
					var = THIS.infamy
					value = 100
					compare = less_than
				}
			}

			infamy_icon_pariah_visible = {
				check_variable = {
					var = THIS.infamy
					value = 100
					compare = greater_than_or_equals 
				}
			}
		}
		
		effects = {
			infamy_bg_click = {
				set_temp_variable = { show = 1 }
				subtract_from_temp_variable = { show = show_fuhrer_powers }
				set_variable = { show_fuhrer_powers = show }
				log = "myVariable: [?THIS.infamy]"
			}
            
			infamy_bg_right_click = {
				add_to_variable = { infamy = 100 }
                set_variable = { infamy = 500 }
				log = "myVariable: [?THIS.infamy]"
			}
		}
	}
    
}
