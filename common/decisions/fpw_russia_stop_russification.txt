	create_poland_university = {
		fire_only_once = yes
		icon = generic_research
		available = {
			
		}
		visible = {
			has_country_flag = russification_poland_visible
			has_completed_focus = SOV_accessibility_of_culture
			
		}
		allowed = {
			
		}
		cost = 80
		complete_effect = {
			
			if = {
				limit = {
					check_variable = { poland_russification > 19 }
				}
				add_to_variable = { poland_russification = -20 }
				custom_effect_tooltip = lower_russification_by_20_tp
			}
			else = {
				
				set_variable = { poland_russification = 0 }
				custom_effect_tooltip = lower_russification_by_20_tp
			}
			
			add_stability = 0.01
			add_war_support = -0.01
			
			if = {
				limit = {
					check_variable = { poland_russification_resist < 91 }
				}
				add_to_variable = { poland_russification_resist = 10 }
				custom_effect_tooltip = raise_resist_by_10_tp
			}
			else = {
				set_variable = { poland_russification_resist = 100 }
				custom_effect_tooltip = raise_resist_by_10_tp
			}
			
			if = {
				limit = {
					check_variable = { poland_russification_coop < 81 }
				}
				add_to_variable = { poland_russification_coop = 20 }
				custom_effect_tooltip = raise_coop_by_20_tp
			}
			else = {
				set_variable = { poland_russification_coop = 100 }
				custom_effect_tooltip = raise_coop_by_20_tp
			}
			
		}
		
	}
