on_actions = {
	on_startup = {
		effect = {
			every_country = {
				set_variable = { THIS.infamy = 0 }
			}
		}
	}

	# FROM is war target.
	# ROOT is for the country who is declaring war 
	on_declare_war = {
		effect = {
			FROM = {
				set_variable = {
					FROM.states = num_owned_controlled_states
				}
				set_variable = {
					FROM.men = manpower_k 
				}
				multiply_variable = {
					FROM.men = FROM.states
				}
			}
			ROOT = {
				set_variable = {
					ROOT.states = num_owned_controlled_states
				}
				set_variable = {
					ROOT.men = manpower_k 
				}
				multiply_variable = { ROOT.men = ROOT.states }
			}
			divide_variable = {
				ROOT.men = FROM.manpower
			}
			add_to_variable = { ROOT.men = 10 }
			add_to_variable = { ROOT.infamy = ROOT.men }
			log = "[?ROOT.men]"
		}
	}
	

	on_liberate = {
		effect = {

		}
	}

	on_puppet = {
		effect = {

		}
	}

	on_release_as_free = {
		effect = {
			
		}
	}

	on_release_as_puppet = {
		effect = {
			
		}
	}

	on_subject_annexed = {
		effect = {
			
		}
	}

	on_generate_wargoal = {
		effect = {
			
		}
	}

	on_justifying_wargoal_pulse = {
		effect = {
			ROOT = {
				add_to_variable = { THIS.infamy = 0.1 }
			}
		}
	}

	# cheating is bad!
	on_nuke_drop = {
		effect = {
			ROOT = {
				add_to_variable = { THIS.infamy = 1000 }
			}
		}
	}

	on_daily_TRB = {
		# decrease infamy
		effect = {
			every_country = {
				add_to_variable = { THIS.infamy = 1 }
			}
		}
	}
}
